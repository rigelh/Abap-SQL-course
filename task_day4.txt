LOOP AT GT_PUNTORET INTO GS_PUNTORET.

ENDIF.

ENDIF.

IF gv_min > GS_PUNTORET-RROGA.
gv_min = GS_PUNTORET-RROGA.
ENDIF.

ENDLOOP.

DELETE GT_PUNTORET WHERE ( rroga = gv_max ).

LOOP AT GT_PUNTORET INTO GS_PUNTORET.

gv_count = 1 + gv_count.
gv_total = gs_puntoret-rroga + gv_total.

ENDLOOP.

gv_mesatare = gv_total / gv_count.

LOOP AT GT_PUNTORET INTO GS_PUNTORET.

IF gs_puntoret-rroga <  gv_mesatare.

ENDIF.
