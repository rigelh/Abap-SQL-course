TYPES: BEGIN OF TY_NR,

         NUMRI1    TYPE P LENGTH 3 DECIMALS 3,
         NUMRI2    TYPE P LENGTH 3 DECIMALS 3,

         MBLEDHJA  TYPE I,
         HEQJA     TYPE P LENGTH 3 DECIMALS 3,
         SHUMEZIMI TYPE P LENGTH 3 DECIMALS 3,
         PJESTIMI  TYPE F,


         FUQI      TYPE P LENGTH 3 DECIMALS 3,
         DATA      TYPE D,
         TIME      TYPE T,

       END OF TY_NR.

DATA:
GS_NR TYPE TY_NR,
LV_NUMRI1 TYPE I,
LV_NUMRI2 TYPE I.


PARAMETERS: P_NUMRI1 TYPE P LENGTH 3 DECIMALS 3,
            P_NUMRI2 TYPE P LENGTH 3 DECIMALS 3.

LV_NUMRI1 = P_NUMRI1.
LV_NUMRI2 = P_NUMRI2.
GS_NR-NUMRI1 = P_NUMRI1.
GS_NR-NUMRI2 = P_NUMRI2.
GS_NR-MBLEDHJA = P_NUMRI1 + P_NUMRI2.
GS_NR-HEQJA = P_NUMRI1 - P_NUMRI2.
GS_NR-SHUMEZIMI = P_NUMRI1 * P_NUMRI2.
GS_NR-PJESTIMI = P_NUMRI1 / P_NUMRI2.

GS_NR-FUQI = P_NUMRI1 ** P_NUMRI2.

GS_NR-DATA =  SY-DATUM + LV_NUMRI1.

GS_NR-TIME =  SY-UZEIT + LV_NUMRI2.


WRITE:
GS_NR-NUMRI1,
GS_NR-NUMRI2,
GS_NR-MBLEDHJA,
GS_NR-HEQJA,
GS_NR-SHUMEZIMI,
GS_NR-PJESTIMI,
GS_NR-FUQI,
GS_NR-DATA,
GS_NR-TIME.
